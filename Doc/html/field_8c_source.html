21 static gboolean
22 on_fielddraw_button_press_event( GtkWidget * widget,
23  GdkEventButton * event,
24  gpointer user_data )
25 {
26  int x,
27  y;
28  GtkAdjustment *hadj,
29  *vadj;
30  GdkCursor *cursor;
31 
32  cursor = gdk_cursor_new( GDK_FLEUR );
33  gdk_window_set_cursor( fielddraw->window, cursor );
34  gdk_cursor_destroy( cursor );
35 
36  /*
37  * Make sure it's a button event
38  */
39  if ( event->type != GDK_BUTTON_PRESS )
40  return FALSE;
41 
42  /*
43  * Get mouse position
44  */
45  x = ( int ) event->x;
46  y = ( int ) event->y;
47  if ( ( x < 0 ) || ( x >= config.life_width ) || ( y < 0 ) || ( y >= config.life_height ) )
48  return FALSE;
49 
50  /*
51  * Do nothing if cursor is already on the hand
52  */
53  zoom_xoffset = zoom_yoffset = 0;
54  if ( ( x >= zoom_xstart ) && ( x < zoom_xstart + zoom_width ) &&
55  ( y >= zoom_ystart ) && ( y < zoom_ystart + zoom_height ) )
56 
57  {
58  zoom_xoffset = x - zoom_xstart;
59  zoom_yoffset = y - zoom_ystart;
60  return FALSE;
61  }
62 
63 #ifdef TRACE_EVENTS
64  TRACE( "field: on_fielddraw_button_press_event: \n" );
65 #endif
66 
67  /*
68  * Move zoom window
69  */
70  hadj = gtk_scrolled_window_get_hadjustment( GTK_SCROLLED_WINDOW( zoom_scrolledwindow ) );
71  vadj = gtk_scrolled_window_get_vadjustment( GTK_SCROLLED_WINDOW( zoom_scrolledwindow ) );
72  gtk_adjustment_set_value( hadj, x );
73  gtk_adjustment_set_value( vadj, y );
74 
75  return FALSE;
76 
77 } // on_fielddraw_button_press_event

178 static void
179 create_hand_field( void )
180 {
181  UBYTE *field,
182  *pw;
183  int row,
184  col;
185  static int prev_zoom_xstart = -1,
186  prev_zoom_ystart;
187  static int prev_zoom_width,
188  prev_zoom_height;
189 
190 
191 #define MAX_POINTS 100
192  static gint on_npoints;
193  static GdkPoint on_points[MAX_POINTS];
194 
195  get_zoom_infos( );
196  field = CURRENT_FIELD;
197 
198  /*
199  * --- Redraw previous zoom area ---
200  */
201  if ( prev_zoom_xstart != -1 )
202 
203  {
204  gdk_draw_rectangle( pixmap_field, field_off_color, TRUE, prev_zoom_xstart,
205  prev_zoom_ystart, prev_zoom_width, prev_zoom_height );
206  on_npoints = 0;
207  for ( row = prev_zoom_ystart; row < prev_zoom_ystart + prev_zoom_height; row++ )
208 
209  {
210  pw = &field[( row * life_bpl ) + ( prev_zoom_xstart / 8 )];
211  for ( col = prev_zoom_xstart; col < prev_zoom_xstart + prev_zoom_width; col++ )
212 
213  {
214  if ( *pw & mask[col % 8] )
215 
216  {
217  on_points[on_npoints].x = col;
218  on_points[on_npoints++].y = row;
219  }
220  if ( col % 8 == 7 )
221  pw++;
222  }
223  }
224  if ( on_npoints )
225  gdk_draw_points( pixmap_field, field_on_color, on_points, on_npoints );
226  }
227 
228  /*
229  * --- X-draw current zoom area ---
230  */
231  gdk_draw_rectangle( pixmap_field, field_on_color, TRUE, zoom_xstart, zoom_ystart,
232  zoom_width, zoom_height );
233  on_npoints = 0;
234  for ( row = zoom_ystart; row < zoom_ystart + zoom_height; row++ )
235  {
236  pw = &field[( row * life_bpl ) + ( zoom_xstart / 8 )];
237  for ( col = zoom_xstart; col < zoom_xstart + zoom_width; col++ )
238 
239  {
240  if ( *pw & mask[col % 8] )
241 
242  {
243  on_points[on_npoints].x = col;
244  on_points[on_npoints++].y = row;
245  }
246  if ( col % 8 == 7 )
247  pw++;
248  }
249  }
250  if ( on_npoints )
251  gdk_draw_points( pixmap_field, field_off_color, on_points, on_npoints );
252  prev_zoom_xstart = zoom_xstart;
253  prev_zoom_ystart = zoom_ystart;
254  prev_zoom_width = zoom_width;
255  prev_zoom_height = zoom_height;
256 
257 } // create_hand_field
